### 利用TELNET验证邮件命令(可以是川大校园邮箱、163邮箱等。)

```python
from pwn import *
import base64
import ssl
context(arch='amd64', os='linux',log_level='debug')
host = 'smtp.163.com'
port = 25

def send_mail():
    #args = ['telnet',host,port]
    context = ssl._create_unverified_context()
    io = remote(host, port,ssl=False,ssl_context=context)
    mail_addr = b'nokknock@163.com'

    io.recvuntil(b'220')
    command1 = b'HELO ' + mail_addr + b'\r'
    io.sendline(command1)
    io.recvuntil(b'250')

    command2 = b'auth login' + b'\r'
    io.sendline(command2)
    io.recvuntil(b'334')

    command3 = base64.encodebytes(b'nokknock@163.com')[0:-1] + b'\r'
    io.sendline(command3)
    io.recvuntil(b'334')

    code = b'LXSAAMwB3pwxBKx7'
    command4 = base64.encodebytes(code)[0:-1] + b'\r'
    io.sendline(command4)
    io.recvuntil(b'235')

    command5 = b'mail from:<' + mail_addr + b'>\r'
    io.sendline(command5)
    io.recvuntil(b'250')

    target = b'tanhuimianbao@163.com'
    command6 = b'rcpt to:<' + target +  b'>\r'
    io.sendline(command6)
    io.recvuntil(b'250')
    io.sendline(b'data')
    io.recvuntil(b'354')
    data = b'''from:nokknock@163.com\r\n'''
    data += b'''to:tanhuimianbao@163.com\r\n'''
    data += b'''subject:BACK AS I NEVER LEFT!\r\n\n'''
    data += b'''BACK AS I NEVER LEFT!\r\n'''
    data += b'''.\r'''
    io.sendline(data)
    io.sendline(b'quit')

if __name__ == '__main__':
    send_mail()
```
